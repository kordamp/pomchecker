018package org.kordamp.maven.checker;
019
020import org.apache.maven.model.Dependency;
021import org.apache.maven.model.Model;
022import org.apache.maven.plugin.logging.Log;
023import org.apache.maven.project.MavenProject;
024
025import java.util.ArrayList;
026import java.util.List;
027
028/**
029 * Checks if a POM file is a minimal BOM file.
030 *
031 * The following blocks are required:
032 * <ul>
033 *   <li>&lt;dependencyManagement&gt;</li>
034 * </ul>
035 *
036 * The following blocks are forbidden:
037 * <ul>
038 *   <li>&lt;build&gt;</li>
039 *   <li>&lt;reporting&gt;</li>
040 *   <li>&lt;dependencies&gt;</li>
041 *   <li>&lt;repositories&gt;</li>
042 *   <li>&lt;pluginRepositories&gt;</li>
043 *   <li>&lt;profiles&gt;</li>
044 *   <li>&lt;modules&gt;</li>
045 * </ul>
046 *
047 * @author Andres Almiray
048 * @since 1.0.0
049 */
050public class BomChecker {
051    /**
052     * Checks the resolved model of the given MaveProject for compliance.
053     *
054     * @param log     the logger to use.
055     * @param project the project to be checked.
056     * @throws PomCheckException if the POM is invalid
057     */
058    public static void check(Log log, MavenProject project) throws PomCheckException {
059        Model model = project.getOriginalModel();
060
061        List<String> errors = new ArrayList<>();
062
063        // 1. is it packaged as 'pom'?
064        if (!"pom".equals(model.getPackaging())) {
065            errors.add("The value of <packaging> must be 'pom'.");
066        }
067
068        // 2. must have a <dependencyManagement> block
069        if (null != model.getDependencyManagement()) {
070            List<Dependency> dependencies = model.getDependencyManagement().getDependencies();
071            if (dependencies == null || dependencies.isEmpty()) {
072                errors.add("No dependencies have been defined in <dependencyManagement>.");
073            }
074        } else {
075            errors.add("No <dependencyManagement> block has been defined.");
076        }
077
078        if (null != model.getBuild()) {
079            errors.add("The <build> block should not be present.");
080        }
081        if (null != model.getReporting()) {
082            errors.add("The <reporting> block should not be present.");
083        }
084        if (null != model.getDependencies() && !model.getDependencies().isEmpty()) {
085            errors.add("The <dependencies> block should not be present.");
086        }
087        if (null != model.getRepositories() && !model.getRepositories().isEmpty()) {
088            errors.add("The <repositories> block should not be present.");
089        }
090        if (null != model.getPluginRepositories() && !model.getPluginRepositories().isEmpty()) {
091            errors.add("The <pluginRepositories> block should not be present.");
092        }
093        if (null != model.getProfiles() && !model.getProfiles().isEmpty()) {
094            errors.add("The <profiles> block should not be present.");
095        }
096        if (null != model.getModules() && !model.getModules().isEmpty()) {
097            errors.add("The <modules> block should not be present.");
098        }
099
100        if (!errors.isEmpty()) {
101            StringBuilder b = new StringBuilder("\nThe POM file\n")
102                .append(project.getFile().getAbsolutePath())
103                .append("\nis not a valid BOM due to the following reasons:\n");
104            for (String s : errors) {
105                b.append(" * ").append(s).append("\n");
106            }
107
108            throw new PomCheckException(b.toString());
109        }
110    }
111}
