018package org.kordamp.maven.checker;
019
020import org.apache.maven.model.Dependency;
021import org.apache.maven.model.Model;
022import org.apache.maven.project.MavenProject;
023
024import java.util.ArrayList;
025import java.util.List;
026
027/**
028 * Checks if a POM file is a minimal BOM file.
029 *
030 * The following blocks are required:
031 * <ul>
032 *   <li>&lt;dependencyManagement&gt;</li>
033 * </ul>
034 *
035 * The following blocks are forbidden:
036 * <ul>
037 *   <li>&lt;build&gt;</li>
038 *   <li>&lt;reporting&gt;</li>
039 *   <li>&lt;dependencies&gt;</li>
040 *   <li>&lt;repositories&gt;</li>
041 *   <li>&lt;pluginRepositories&gt;</li>
042 *   <li>&lt;profiles&gt;</li>
043 *   <li>&lt;modules&gt;</li>
044 * </ul>
045 *
046 * @author Andres Almiray
047 * @since 1.0.0
048 */
049public class BomChecker {
050    /**
051     * Checks the resolved model of the given MaveProject for compliance.
052     *
053     * @param log     the logger to use.
054     * @param project the project to be checked.
055     * @throws PomCheckException if the POM is invalid
056     */
057    public static void check(Logger log, MavenProject project) throws PomCheckException {
058        Model model = project.getOriginalModel();
059
060        List<String> errors = new ArrayList<>();
061
062        // 1. is it packaged as 'pom'?
063        if (!"pom".equals(model.getPackaging())) {
064            errors.add("The value of <packaging> must be 'pom'.");
065        }
066
067        // 2. must have a <dependencyManagement> block
068        if (null != model.getDependencyManagement()) {
069            List<Dependency> dependencies = model.getDependencyManagement().getDependencies();
070            if (dependencies == null || dependencies.isEmpty()) {
071                errors.add("No dependencies have been defined in <dependencyManagement>.");
072            }
073        } else {
074            errors.add("No <dependencyManagement> block has been defined.");
075        }
076
077        if (null != model.getBuild()) {
078            errors.add("The <build> block should not be present.");
079        }
080        if (null != model.getReporting()) {
081            errors.add("The <reporting> block should not be present.");
082        }
083        if (null != model.getDependencies() && !model.getDependencies().isEmpty()) {
084            errors.add("The <dependencies> block should not be present.");
085        }
086        if (null != model.getRepositories() && !model.getRepositories().isEmpty()) {
087            errors.add("The <repositories> block should not be present.");
088        }
089        if (null != model.getPluginRepositories() && !model.getPluginRepositories().isEmpty()) {
090            errors.add("The <pluginRepositories> block should not be present.");
091        }
092        if (null != model.getProfiles() && !model.getProfiles().isEmpty()) {
093            errors.add("The <profiles> block should not be present.");
094        }
095        if (null != model.getModules() && !model.getModules().isEmpty()) {
096            errors.add("The <modules> block should not be present.");
097        }
098
099        if (!errors.isEmpty()) {
100            StringBuilder b = new StringBuilder("\nThe POM file\n")
101                .append(project.getFile().getAbsolutePath())
102                .append("\nis not a valid BOM due to the following reasons:\n");
103            for (String s : errors) {
104                b.append(" * ").append(s).append("\n");
105            }
106
107            throw new PomCheckException(b.toString());
108        }
109    }
110}
