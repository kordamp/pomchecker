018package org.kordamp.maven.checker.cli.internal;
019
020import picocli.CommandLine;
021
022import java.io.PrintWriter;
023
024/**
025 * @author Andres Almiray
026 * @since 1.1.0
027 */
028public class Colorizer extends PrintWriter {
029    public Colorizer(PrintWriter delegate) {
030        super(delegate, true);
031    }
032
033    @Override
034    public void print(String s) {
035        super.print(CommandLine.Help.Ansi.AUTO.text("@|red " + s + "|@"));
036    }
037}
