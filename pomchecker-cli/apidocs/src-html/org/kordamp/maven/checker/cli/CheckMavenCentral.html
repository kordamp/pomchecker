018package org.kordamp.maven.checker.cli;
019
020import org.apache.maven.project.MavenProject;
021import org.kordamp.maven.checker.MavenCentralChecker;
022import org.kordamp.maven.checker.PomCheckException;
023import org.kordamp.maven.checker.cli.internal.PomParser;
024import picocli.CommandLine;
025
026/**
027 * @author Andres Almiray
028 * @since 1.1.0
029 */
030@CommandLine.Command(name = "check-maven-central",
031    description = "Checks if a POM complies with the rules for uploading to Maven Central")
032public class CheckMavenCentral extends AbstractCommand {
033    @CommandLine.Option(names = {"--no-strict"},
034        description = "Checks if <repositories> and <pluginRepositories> are present")
035    boolean strict;
036
037    @CommandLine.Option(names = {"--no-release"},
038        description = "Checks if version is -SNAPSHOT")
039    boolean release;
040
041    @Override
042    protected void execute() {
043        try {
044            logger.info("Maven Central checks: {}", pomFile.toAbsolutePath().toString());
045            MavenProject project = PomParser.createMavenProject(pomFile.toFile());
046            MavenCentralChecker.check(logger, project, !release, !strict);
047        } catch (PomCheckException e) {
048            throw new HaltExecutionException(e);
049        }
050    }
051}
